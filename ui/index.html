// Handle inventory updates from chat messages hytopia.onData((data) => { if (data.type === "inventory_update" && data.inventory) { updateInventoryUI(data.inventory, data.heldItemIndex); } }); function updateInventoryUI(inventory, heldItemIndex) { document.querySelectorAll(".inventory-slot").forEach((slot, index) => { const item = inventory[index]; const isHeld = index === heldItemIndex; // Reset all slots first slot.classList.remove("selected", "empty", "has-item"); if (item) { // If there's an item, show its name slot.textContent = item; slot.classList.add("has-item"); if (isHeld) { // If this is the held item, add selected class slot.classList.add("selected"); } } else { // If no item, show the slot number slot.textContent = (index + 1).toString(); slot.classList.add("empty"); } }); }
